78 void Sprite::scaleTexture() {

80  float ratioH = transform.size.y / _texture->getHeight();
81 
82  auto& uv = _geometry->getAttribute("Texture").getVector();
83  for (unsigned int i = 0; i < uv.size(); i+=2) {
84  if (uv[i] != 0)
85  uv[i] = ratioW;
